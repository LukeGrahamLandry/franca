
fn cli_args() Ptr(Str);

const CmdResult: Type;
fn system(cmd: Ptr(Str)) CmdResult;

fn system_no_args(cmd: Str) CmdResult = { 
    let parts = (cmd)!slice;
    let parts: List(Str) = list_clone(parts);   // TODO: Vec<String>::deserialize can't deal with stack addrs. 
    let parts: Ptr(Str) = items(parts!addr);
    system(parts)
}
